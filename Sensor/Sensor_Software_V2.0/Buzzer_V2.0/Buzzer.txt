

const int buzzer = 10; // Buzzer auf pin 10


void setup(){
 
  pinMode(buzzer, OUTPUT); // Setzen Buzzer auf pin 10 als Ausgang

}

void loop(){
 
  tone(buzzer, 1000); // 1kHz Tonsignal 
  delay(1000);        // 1 Sekunde warten
  noTone(buzzer);     // Stop Ton
  delay(1000);        // 1 Sekunde warten
  
}
